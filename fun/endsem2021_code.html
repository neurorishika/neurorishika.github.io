df= pd.concat([one,two]).reset_index(drop=True)
df = df.drop(['EMAIL','MEMBER ID'],axis=1)
df = df.drop([8,266]).reset_index(drop=True)
df["MAIN SUBJECT"] = df['GROUP NAMES'].str.split(',').apply(lambda v: np.unique([k.split('_')[1][:3] for k in v])[np.unique([k.split('_')[1][:3] for k in v],return_counts=True)[1].argmax()])
ent = lambda v: np.sum([-k*np.log(k) for k in v])
df['INTERDISCIPLINARITY'] = df['GROUP NAMES'].str.split(',').apply(lambda v: ent(np.unique([k.split('_')[1][:3] for k in v],return_counts=True)[1]/len(v)))
df['NUMBER OF COURSES'] = df["GROUP NAMES"].str.split(',').apply(lambda v:len(v))
df['SPECIALIZATION'] = (1-(df['INTERDISCIPLINARITY']/ent([1/6,1/6,1/6,1/6,1/6,1/6])))
df['SUBJECTS'] = df["GROUP NAMES"].str.split(',').apply(lambda v:",".join(np.unique([k.split('_')[1][:3] for k in v])))
nmap = {v: k for k, v in dict(df.NAME).items()}
conn = np.zeros((df.shape[0],df.shape[0]))

ex = df.assign(COURSE=df['GROUP NAMES'].str.split(',')).drop("GROUP NAMES",axis=1).explode('COURSE').reset_index(drop=True)

for i in ex['COURSE'].unique():
    peeps = list(ex[ex['COURSE']==i]['NAME'])
    index = [nmap[k] for k in peeps]
    for x,y in combinations(index,2):
        conn[x,y] += 1
        conn[y,x] = conn[x,y]

df['COMMUNITY SIZE'] = pd.Series(np.sum(conn>0,axis=0))
df['COMMUNITY STRENGTH'] = pd.Series([np.mean(k[k>0]) for k in conn])
df= df.drop("GROUP NAMES",axis=1)

G = nx.from_numpy_array(conn, create_using=nx.Graph)

subcode = {"BIO":2,
"ECS":1,
"PHY":0,
"CHM":3,
"MTH":4,
"HSS":5}

nx.set_node_attributes(G, df["NAME"], "Name")
nx.set_node_attributes(G, df["MAIN SUBJECT"], "Subject")
nx.set_node_attributes(G, df["BATCH"], "batch")
nx.set_node_attributes(G, df["NUMBER OF COURSES"], "ncourses")
nx.set_node_attributes(G, df["SPECIALIZATION"], "Specialization")
nx.set_node_attributes(G, 10*df["SPECIALIZATION"]+5, "nodesize")
nx.set_node_attributes(G, df["MAIN SUBJECT"].apply(lambda v: subcode[v]), "subcode")
nx.set_node_attributes(G, df['COMMUNITY SIZE'], "comsize")
nx.set_node_attributes(G, df['COMMUNITY STRENGTH'], "comstr")
nx.set_node_attributes(G, df['SUBJECTS'], "subs")

from bokeh.io import output_notebook, show, save
from bokeh.models import Range1d, Circle, ColumnDataSource, MultiLine
from bokeh.plotting import figure
from bokeh.plotting import from_networkx
from bokeh.palettes import Spectral6,Greys5
from bokeh.transform import linear_cmap
from bokeh.models import EdgesAndLinkedNodes, NodesAndLinkedEdges
output_notebook()

title = 'IISER End-Sem Registration (Jan 2021)'

HOVER_TOOLTIPS = [("Student", "@Name"),("Main Subject", "@Subject"),("Batch", "@batch"),("Number of Courses", "@ncourses"),("Specialization Score", "@Specialization"),
                 ("Community Size", "@comsize"),("Community Strength", "@comstr"),("Subjects Taken:", "@subs")]


#Create a plot — set dimensions, toolbar, and title
plot = figure(tooltips = HOVER_TOOLTIPS,
              tools="pan,wheel_zoom,save,reset", active_scroll='wheel_zoom',
            x_range=Range1d(-10.1, 10.1), y_range=Range1d(-10.1, 10.1), title=title,
             plot_width=800, plot_height=800)

network_graph = from_networkx(G, nx.fruchterman_reingold_layout, scale=10, center=(0, 0))

#Set node size and color
color_by_this_attribute = "subcode"
minimum_value_color = np.min(network_graph.node_renderer.data_source.data[color_by_this_attribute])
maximum_value_color = np.max(network_graph.node_renderer.data_source.data[color_by_this_attribute])
network_graph.node_renderer.glyph = Circle(size='nodesize', line_color = linear_cmap('batch', Greys5, 2017, 2019),line_width = 1,
                                           fill_color=linear_cmap(color_by_this_attribute, Spectral6, minimum_value_color, maximum_value_color))
network_graph.node_renderer.hover_glyph = Circle(size='nodesize', fill_color='white', line_width=2)
network_graph.node_renderer.selection_glyph = Circle(size='nodesize', fill_color='white', line_width=2)

#Set edge opacity and width
network_graph.edge_renderer.glyph = MultiLine(line_alpha=0.5, line_width=0.02)
network_graph.edge_renderer.selection_glyph = MultiLine(line_color='black', line_width=1)
network_graph.edge_renderer.hover_glyph = MultiLine(line_color='black', line_width=1)

network_graph.selection_policy = NodesAndLinkedEdges()
network_graph.inspection_policy = NodesAndLinkedEdges()

#Add network graph to the plot
plot.renderers.append(network_graph)


show(plot)
save(plot, filename=f"E:/Personal Projects/Personal Website/neurorishika.github.io/endsem2021.html",title=title)
